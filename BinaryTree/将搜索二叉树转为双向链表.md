# 将搜索二叉树转为双向链表

```java
/**
 *              4
 *          2      6
 *        1  3    5 7
 */
public class BSTCovertToLinkedList {
    public static TreeNode covert(TreeNode root) {
        if (root == null) {
            return null;
        }

        TreeNode cur = root;
        TreeNode pre = null;
        TreeNode mostRight = null;

        while (cur != null) {
            mostRight = cur.left;
            if (mostRight != null) {
                while (mostRight.right != null && mostRight.right != cur) {
                    mostRight = mostRight.right;
                }
                if (mostRight.right == null) {
                    mostRight.right = cur;
                    cur = cur.left;
                    continue;
                }
            }
            if (pre != null) {
                pre.right = cur;
                cur.left = pre;
            }
            pre = cur;
            cur = cur.right;
        }

        TreeNode head = pre;
        while (head.left != null) {
            head = head.left;
        }

        return head;
    }

    public static void main(String []args) {
        TreeNode head = covert(BinaryTreeUtil.createBinaryTree5());
        TreeNode tail = head;
        while (head != null) {
            System.out.print(head.val + " ");
            tail = head;
            head = head.right;
        }

        System.out.print("\n");
        while (tail != null) {
            System.out.print(tail.val + " ");
            tail = tail.left;
        }
    }
}
```
